Script Java


ALGORITMO 66

        
            Algoritmo "Calculadora de Fatorial" 66

    Var
     numero, fatorial, contador: inteiro

Inicio
   Escreva("Digite um número inteiro: ")
   Leia(numero)

   fatorial <- 1
   contador <- numero

   Enquanto contador > 1 faca
      fatorial <- fatorial * contador
      contador <- contador - 1
   fimenquanto

   Escreva("O fatorial de ", numero, " é ", fatorial)
